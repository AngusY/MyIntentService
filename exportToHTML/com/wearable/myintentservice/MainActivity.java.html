67       public class MyReceiver extends BroadcastReceiver { 
68    
69           @Override 
70           public void onReceive(Context context, Intent intent) { 
71               Log.v(TAG, "OnReceive() called."); 
72    
73               final String action = intent.getAction(); 
74               if (ACTION_BROADCAST_FOO.equals(action)) { 
75                   String foo_out_msg = intent.getStringExtra("foo_out"); 
76                   TextView fooTV = (TextView) findViewById(R.id.foo_tv); 
77                   if (fooTV != null) fooTV.setText(foo_out_msg); 
78    
79               } else if (ACTION_BROADCAST_BAZ.equals(action)) { 
80                   String baz_out_msg = intent.getStringExtra("baz_out"); 
81                   TextView bazTV = (TextView) findViewById(R.id.baz_tv); 
82                   if (bazTV != null) bazTV.setText(baz_out_msg); 
83    
84               } 
85           } 
86       }
